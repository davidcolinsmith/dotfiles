# From https://shapeshed.com/vim-packages/

# Add vim plugin
git submodule init
git submodule add \
  https://github.com/fancy-plugin/fancy-plugin.git \
  vim/pack/[look|syntax|function]/[start|opt]/fancy-plugin
git add vim/pack/[look|syntax|function]/[start|opt]/fancy-plugin
git commit

# Update plugin
git submodule update --remote --merge
git commit

# Remove plugin
git submodule deinit vim/pack/[look|syntax|function]/[start|opt]/fancy-plugin
git rm vim/pack/[look|syntax|function]/[start|opt]/fancy-plugin
rm -rf vim/pack/[look|syntax|function]/[start|opt]/fancy-plugin
git commit
